## Get all bills

Let's imagine you want to sync all your revenue from Proper to your own accounting system.

<CodeGroup>

```bash cURL
curl --request GET \
  --url https://api.helloproper.com/bills \
  --header 'Authorization: your_secret_key
```

```javascript JavaScript
const axios = require('axios');
const fs = require('fs');
const yourSecretKey = 'your_secret_key'; // Replace with your actual secret key
const apiEndPoint = 'https://api.helloproper.com/bills';

let config = {
  method: 'get',
  url: apiEndPoint,
  headers: {
    Authorization: yourSecretKey,
  },
};
let results = [];

(async function loop(nextPageUrl = '') {
  try {
    // If there's a next page URL provided from the last request, use it, otherwise, use the initial API endpoint
    config.url = nextPageUrl || apiEndPoint;
    const response = await axios(config);

    // Push the bills from the current page to the results array
    results.push(...response.data.bills);

    // Check if there's a next page. This assumes the API provides a next page URL if there are more bills to fetch
    if (response.data.nextPageUrl) {
      loop(response.data.nextPageUrl); // Recursively call the loop function with the new page URL
    } else {
      // If there are no more pages, save the results to a file
      fs.writeFileSync('bills.json', JSON.stringify(results));
      console.log('All bills have been fetched and saved to bills.json');
    }
  } catch (error) {
    console.error('An error occurred:', error);
  }
})();

```
