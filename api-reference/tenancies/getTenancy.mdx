---
title: "Retrieve a tenancy"
api: "GET /tenancies/{tenancyId}"
---

<ParamField path="tenancyId" type="string" required>
  Tenancy id
</ParamField>

  </ResponseField>

      <ResponseField name="floor" type="string">

      </ResponseField>

      <ResponseField name="door" type="string">
        The door of the address if applicable
