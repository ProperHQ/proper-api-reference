---
title: "List tenancies"
api: "GET /tenancies"
---

### Response

<ResponseField name="id" type="string">
  The unique identifier for the tenancy.
</ResponseField>

<ResponseField name="address" type="Address Object">
  The address of the tenancy.
</ResponseField>

<ResponseField name="currentLease" type="ResponseLease Object (optional)">
  Details of the current lease, if available.
</ResponseField>

<ResponseField name="leaseIds" type="Array of string">
  List of lease IDs associated with the tenancy.
</ResponseField>

<ResponseField name="unitType" type="UnitType (optional)">
  The type of unit (e.g., APARTMENT, HOUSE).
</ResponseField>

<ResponseField name="unitTypeName" type="string (optional)">
  The name of the unit type.
</ResponseField>

<ResponseField name="roomCounts" type="Array of RoomCount Objects (optional)">
  The count of various types of rooms in the tenancy.
</ResponseField>

<ResponseField name="totalArea" type="ResponseArea Object (optional)">
  Total area of the tenancy.
</ResponseField>

<ResponseField name="businessArea" type="ResponseArea Object (optional)">
  Business area of the tenancy.
</ResponseField>

<ResponseField name="archivedAt" type="ZonedDateTime (optional)">
  The date and time when the tenancy was archived.
</ResponseField>
