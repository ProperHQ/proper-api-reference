---
title: "List payouts"
api: "GET /payouts"
---

### Request

This request supports pagination.


### Response

<ResponseField name="payouts" type="Array of Payout Object">
  <Expandable title="properties of Payout Objects">

    <ResponseField name="id" type="string">
      The unique identifier for the payout
    </ResponseField>

    <ResponseField name="payoutReason" type="string">
      The reason for the payout. Can be either Payment or Refund
    </ResponseField>

    <ResponseField name="payoutDate" type="datetime">
      The date the payout was made
    </ResponseField>

    <ResponseField name="net" type="Amount Object">
       The net amount of the payout

       <Expandable title="properties of Amount Object">

         <ResponseField name="currency" type="string">
           The currency code for the net amount (e.g., "DKK")
         </ResponseField>

         <ResponseField name="amount" type="number">
           The amount of the net amount (e.g., 100.00)
         </ResponseField>

       </Expandable>
    </ResponseField>

    <ResponseField name="fees" type="Amount Object">
       The fees subtracted from the payout

       <Expandable title="properties of Amount Object">

         <ResponseField name="currency" type="string">
           The currency code for the fees amount (e.g., "DKK")
         </ResponseField>

         <ResponseField name="amount" type="number">
           The amount of the fees amount (e.g., 100.00)
         </ResponseField>

       </Expandable>
    </ResponseField>

    <ResponseField name="gross" type="Amount Object">
       The gross amount of the payout

       <Expandable title="properties of Amount Object">

         <ResponseField name="currency" type="string">
           The currency code for the gross amount (e.g., "DKK")
         </ResponseField>

         <ResponseField name="amount" type="number">
           The amount of the gross amount (e.g., 100.00)
         </ResponseField>

       </Expandable>
    </ResponseField>

    <ResponseField name="bankStatement" type="string">
      The bank statement reference for the payout
    </ResponseField>

  </Expandable>
</ResponseField>
