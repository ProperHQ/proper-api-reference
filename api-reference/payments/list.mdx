---
title: "List payments"
api: "GET /payments"
---

### Request

This request supports pagination.

### Response

<ResponseField name="payments" type="Array of Payment Objects">
  <Expandable title="properties of Payment Object">

    <ResponseField name="id" type="string">
      The unique identifier for the payment
    </ResponseField>

    <ResponseField name="paymentDate" type="datetime">
      The date the payment was made
    </ResponseField>


    <ResponseField name="net" type="Amount Object">
       The net amount of the payment

       <Expandable title="properties of Amount Object">

         <ResponseField name="currency" type="string">
           The currency code for the net amount (e.g., "DKK")
         </ResponseField>

         <ResponseField name="amount" type="number">
           The amount of the net amount (e.g., 100.00)
         </ResponseField>

       </Expandable>
    </ResponseField>

    <ResponseField name="fees" type="Amount Object">
       The fees subtracted from the payment

       <Expandable title="properties of Amount Object">

         <ResponseField name="currency" type="string">
           The currency code for the fees amount (e.g., "DKK")
         </ResponseField>

         <ResponseField name="amount" type="number">
           The amount of the fees amount (e.g., 100.00)
         </ResponseField>

       </Expandable>
    </ResponseField>

    <ResponseField name="gross" type="Amount Object">
       The gross amount of the payment

       <Expandable title="properties of Amount Object">

         <ResponseField name="currency" type="string">
           The currency code for the gross amount (e.g., "DKK")
         </ResponseField>

         <ResponseField name="amount" type="number">
           The amount of the gross amount (e.g., 100.00)
         </ResponseField>

       </Expandable>
    </ResponseField>

    <ResponseField name="payoutId" type="String">
      The payout that resulted from this payment
    </ResponseField>

    <ResponseField name="leaseId" type="String">
      The lease identifier that this payment was made to
    </ResponseField>

  </Expandable>
</ResponseField>
