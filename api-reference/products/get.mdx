---
title: "List products"
api: "GET /products"
---

### Response

<ResponseField name="productKey" type="string">
  The key or identifier for the product.
</ResponseField>

<ResponseField name="productId" type="string (optional)">
  The unique identifier for the product.
</ResponseField>

<ResponseField name="productName" type="Localised String Object">
  The name of the product in different languages.
</ResponseField>

<ResponseField name="archived" type="boolean">
  Indicates whether the product is archived.
</ResponseField>

<ResponseField name="custom" type="boolean">
  Indicates whether the product is a custom product.
</ResponseField>

<ResponseField name="position" type="number">
  The position of the product in the list.
</ResponseField>

<ResponseField name="isDefaultProduct" type="boolean">
  Indicates whether the product is a default product.
</ResponseField>
