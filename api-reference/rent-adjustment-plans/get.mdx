---
title: "List rent adjustment plans"
api: "GET /rent-adjustment-plans"
---

### Response

<ResponseField name="id" type="string">
  The unique identifier for the rent adjustment plan.
</ResponseField>

<ResponseField name="leaseId" type="string">
  The identifier for the lease associated with the rent adjustment plan.
</ResponseField>

<ResponseField name="created" type="ZonedDateTime">
  The creation date and time of the rent adjustment plan.
</ResponseField>

<ResponseField name="modified" type="ZonedDateTime">
  The last modification date and time of the rent adjustment plan.
</ResponseField>

<ResponseField name="adjusteeRef" type="EntityRef">
  The reference to the entity that the rent adjustment plan applies to.
</ResponseField>

<ResponseField name="balanceChargeType" type="SmartText (optional)">
  The type of charge for the balance.
</ResponseField>

<ResponseField name="nextAdjustmentDate" type="LocalDate (optional)">
  The date of the next adjustment.
</ResponseField>

<ResponseField name="cancelledOn" type="LocalDate (optional)">
  The date when the rent adjustment plan was cancelled.
</ResponseField>

<ResponseField name="endedOn" type="LocalDate (optional)">
  The date when the rent adjustment plan ended.
</ResponseField>

<ResponseField name="cancelledOrEndedOn" type="LocalDate (optional)">
  The date when the rent adjustment plan was either cancelled or ended.
</ResponseField>

<ResponseField name="adjustsRecurringBillItem" type="boolean">
  Indicates whether the rent adjustment plan adjusts recurring bill items.
</ResponseField>

<ResponseField name="adjustsBalance" type="boolean">
  Indicates whether the rent adjustment plan adjusts balances.
</ResponseField>

<ResponseField name="vat" type="Tax (optional)">
  VAT details for the rent adjustment plan.
</ResponseField>

<ResponseField name="rentAdjustmentRules" type="Array of RentAdjustmentRule Objects">
  The rules that define how the rent is adjusted.
</ResponseField>

<ResponseField name="adjustments" type="Array of Adjustment Objects">
  The adjustments that have been made according to the rent adjustment plan.
</ResponseField>
